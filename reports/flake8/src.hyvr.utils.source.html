56 def is_point_inside_ellipsoid(
57 x: np.ndarray,
58 y: np.ndarray,
59 z: np.ndarray,
60 x_center: float,
61 y_center: float,
62 z_center: float,
63 a: float,
64 b: float,
65 c: float,
66 alpha: float,
67 ) -> np.ndarray:
68 """
69 Return boolean array with true indices corresponding to points inside the ellipsoid
70 Parameters
71 ----------
72 x,y,z: point coordinates array
73 x_center,y_center,z_center: ellipsoid center
74 a,b,c: major, minor and vertical axis of the ellipsoid
75 alpha: azimuth in counterclockwise from east and in radians
76 """
77 matrix = rotation_matrix_z(alpha)
78 x = np.ravel(x)
79 y = np.ravel(y)
80 z = np.ravel(z)
81 dx = x - x_center
82 dy = y - y_center
83 dz = z - z_center
84 rotated_points = matrix @ np.vstack((dx, dy, dz))
85 distance_vector = (
86 rotated_points[0, :] ** 2 / a**2
87 + rotated_points[1, :] ** 2 / b**2
88 + rotated_points[2, :] ** 2 / c**2
89 )
90 logic = np.where(distance_vector <= 1, True, False)
91 logic = np.reshape(logic, x.shape)
92 return logic

95 @numba.njit()
96 def dip_dip_dir_bulbset(
97 x: np.ndarray,
98 y: np.ndarray,
99 z: np.ndarray,
100 x_center: float,
101 y_center: float,
102 z_center: float,
103 a: float,
104 b: float,
105 c: float,
106 alpha: float,
107 dip: float,
108 ) -> np.ndarray:
109 """
110 Return dip and dip_direction_arrays of the points inside ellipsoid according to the bulb method
111 Parameters
112 ----------
113 x,y,z: point coordinates array
114 x_center,y_center,z_center: ellipsoid center
115 a,b,c: major, minor and vertical axis of the ellipsoid
116 alpha: azimuth in counterclockwise from east and in radians
117 """
118 # Since ellipsoids are isosurfaces of quadratic functions, we can
119 # get the normal vector by taking the gradient of the quadratic
120 # function that has our ellipsoid as iso-surface.
121 # This function can be written as:
122 #
123 # f(d(x)) = (x*cos+y*sin)**2/a**2 + (-x*sin+y*cos)**2/b** + z**2/c**2
124 #
125 # The gradient is (up to a scalar)
126 #
127 # / nx*cos/a + ny*sin/b \
128 # grad f(x) = | -nx*sin/a + ny*cos/b |
129 # \ nz/c /
130 #
131 # where nx, ny, nz are the normalized distances.
132 # The gradient points outwards.
133 # The length of the vector is
134 #
135 # |grad f(x)| = (nx/a)**2 + (ny/b)**2 + (nz/c)**2
136 #
137 # The dip angle is the the angle between the normal
138 # vector and the unit z-vector.
139 # The azimuth is the angle between the projection of the normal
140 # vector onto the x-y-plane and the unit x-vector.
141 dx = x - x_center
142 dx = np.ravel(dx)
143 dy = y - y_center
144 dy = np.ravel(dy)
145 dz = z - z_center
146 dz = np.ravel(dz)
147 cos_alpha = np.cos(alpha)
148 sin_alpha = np.sin(alpha)
149 normalized_dx = (dx * cos_alpha + dy * sin_alpha) / a
150 normalized_dy = (-dx * sin_alpha + dy * cos_alpha) / b
151 normalized_dz = dz / c
152 aaa = normalized_dx / a
153 bbb = normalized_dy / b
154 ccc = normalized_dz / c
155 len_normvec = np.sqrt(aaa**2 + bbb**2 + ccc**2)
156 normvec_x = (aaa * cos_alpha + bbb * sin_alpha) / len_normvec
157 normvec_y = (-aaa * sin_alpha + bbb * cos_alpha) / len_normvec
158 normvec_z = ccc / len_normvec
159  
160 len_normvec_xy = np.sqrt(normvec_x**2 + normvec_y**2)
161 norm_distance = np.sqrt(normalized_dx**2 + normalized_dy**2 + normalized_dz**2)
162  
163 # The dip angle can be found as acos of the normalized
164 # scalar product of unit z-vector and normal vector
165 # It should be negative, if the gradient points in positive
166 # x direction and in positive z direction, or if it points
167 # in negative x and z direction.
168 # direction.
169 dip_bulb = (
170 -np.arccos(np.abs(normvec_z)) * np.sign(normvec_x * normvec_z) / np.pi * 180
171 )
172  
173 # The azimuth angle should also be between -90 and 90. It is
174 # the angle between the projection of the normal vector into
175 # the xy-plane and the x unit vector.
176 # It can be calculated as negative atan of the y component
177 # over the x component. To make sure the angle is between
178 # -90 and 90, we change the sign of both components, if the
179 # x component is negative. This means the x component is
180 # always positive, and the y component is multiplied by the
181 # sign of the x component
182 dip_dir_counterclockwise = (
183 -np.arctan2(np.sign(normvec_x) * normvec_y, np.abs(normvec_x)) / np.pi * 180
184 )
185 dip_dir_bulb = (-dip_dir_counterclockwise + 90) % 360
186  
187 # if len_normvec != 0:
188 # the point is exactly at the center of the trough, this means
189 # there is no azim and dip
190 dip_dir_output = np.where(len_normvec < 1e-12, 0.0, dip_dir_bulb)
191 dip_output = np.where(len_normvec < 1e-10, 0.0, dip_bulb)
192  
193 # normal vector points in z-direction -> dip = 0, azim = 0
194 dip_dir_output = np.where(len_normvec_xy < 1e-12, 0.0, dip_dir_output)
195 dip_output = np.where(len_normvec_xy < 1e-12, 0.0, dip_output)
196  
197 # for geological applications, a negative dip does not make sense:
198 dip_dir_output = np.where(
199 dip_output < 0,
200 np.where(dip_dir_output < 180, dip_dir_output + 180, dip_dir_output - 180),
201 dip_dir_output,
202 )
203 dip_output = np.abs(dip_output)
204 # cap dip in bulb or bulbsets: use dip as maximum dip
205 dip_output = np.where(dip_output > dip, dip, dip_output)
206  
207 return dip_output, dip_dir_output, norm_distance

279 def ferguson_theta_ode(s_max, eps_factor, k, h, omega):
280 """
281 Implementation of (Ferguson, 1976, Eq.9).
282 The equation is formulated as an initial value problem and integrated with scipy function for integration (solve_ivp)
283 http://onlinelibrary.wiley.com/doi/10.1002/esp.3290010403/full
284  
285 Parameters:
286 s_max: max length of channel
287 k: Wavenumber
288 h: Height
289 eps_factor: Random background noise (normal variance)
290 omega: initial angle
291  
292 Returns:
293 theta : angle array
294 s : channel distance array
295  
296 """
297  
298 # Correlated variance calculation => Gaussian function
299 s_range = np.arange(0, s_max, (s_max / 1000))
300 dist_arr = scipy.spatial.distance.pdist(
301 np.expand_dims(s_range, axis=1), "sqeuclidean"
302 )
303 variance = eps_factor
304 cov = variance * np.exp(-(1 / 2) * dist_arr)
305  
306 cov = scipy.spatial.distance.squareform(cov)
307 cov[np.diag_indices_from(cov)] = variance
308 u = np.random.multivariate_normal(
309 np.zeros_like(s_range), np.eye((s_range).shape[0])
310 )
311 L = scipy.linalg.cholesky(cov)
312 e_s = L @ u
313  
314 def rhs(t, y, k, h):
315 eps_t = np.interp(np.array([t]), s_range, e_s)
316 eps_t = eps_t[0] # from array to float
317 d_tau_ds = (eps_t - y[1] - 2 * h / k * y[0]) * (k**2)
318 d_theta_ds = y[0]
319 dx_ds = np.cos(y[1])
320 dy_ds = np.sin(y[1])
321 return np.array([d_tau_ds, d_theta_ds, dx_ds, dy_ds])
322  
323 def jac(t, y, k, h):
324 return np.array(
325 [
326 [-2 * h * k, -(k**2), 0, 0],
327 [1, 0, 0, 0],
328 [0, -np.sin(y[1]), 0, 0],
329 [0, np.cos(y[1]), 0, 0],
330 ]
331 )
332  
333 y0 = np.array([omega * k, 0.0, 0.0, 0.0])
334  
335 solution = scipy.integrate.solve_ivp(
336 rhs,
337 (0, s_max),
338 y0,
339 method="BDF",
340 args=(k, h),
341 first_step=0.01,
342 jac=jac,
343 atol=1e-8,
344 rtol=1e-8,
345 )
346  
347 y = solution.y
348  
349 s = solution.t
350  
351 theta = y[1, :]
352 x = y[2, :]
353 y = y[3, :]
354  
355 return theta, s, x, y

361 def specsim(
362 x: np.array,
363 y: np.array,
364 z=np.array([0.0]),
365 mean=0.0,
366 var=1.0,
367 corl=np.array([1.0, 1.0]),
368 z_axis=0,
369 mask=None,
370 covmod="gaussian",
371 ):
372 """
373 Generate random variables with stationary covariance function using spectral
374 techniques of Dietrich & Newsam (1993)
375  
376 Parameters
377 ----------
378 x,y,z : 3D np.ndarray with x,y,z, coordinates of the grid.
379 mean: mean surface z (or 3d variable)
380 var : float
381 variance of the gaussian value.
382 corl : tuple of floats
383 Tuple of correlation lengths. 2-tuple for 2-d (x and y), and 3-tuple
384 for 3-d (x, y, z).
385 z_axis: int
386 location of the z_axis in the numpy grid.
387 The refernce value is axis 0 (the first), as in the MODFLOW convention.
388 mask: np.array optional:
389 adds a mask to calculate the gaussian random field in a subset of the grid: x[mask]
390 covmod : str, optional (default: "gaussian")
391 Which covariance model to use ("gaussian" or "exp").
392  
393 Returns
394 -------
395 Y : 1d, 2d, or 3d numpy array
396 Numpy array of random field. If no selection mask was given, this is either a 2d or 3d numpy
397 array, depending on ``two_dim`` and of the size of the model grid.
398 If a selection_mask was given, this is a flat array of values.
399 """
400 if covmod not in ["gaussian", "exp"]:
401 raise ValueError("covariance model must be 'gaussian' or 'exp'")
402 if mask is None:
403 x_calc = x
404 y_calc = y
405 else:
406 x_calc = np.where(mask, x, np.nan)
407 y_calc = np.where(mask, y, np.nan)
408 two_dim = len(corl) < 3 # boolean weather calculations should be done in two or 3D
409 if two_dim:
410 Y = np.empty(x.shape)
411 h_square = (x_calc / corl[0]) ** 2 + (y_calc / corl[1]) ** 2
412 else:
413 if mask is None:
414 z_calc = z
415 else:
416 z_calc = np.where(mask, z, z_calc)
417 Y = np.empty(z.shape)
418 h_square = (
419 (x_calc / corl[0]) ** 2 + (y_calc / corl[1]) ** 2 + (z_calc / corl[2]) ** 2
420 )
421 ntot = h_square.size
422 # Covariance matrix of variables
423 if covmod == "gaussian":
424 # Gaussian covariance model
425 ryy = np.exp(-h_square) * var
426 else: # covmod == 'exp':
427 # Exponential covariance model
428 ryy = np.exp(-np.sqrt(h_square)) * var
429 # Power spectrum of variable
430 syy = np.fft.fftn(np.fft.fftshift(ryy)) / ntot
431 syy = np.abs(syy) # Remove imaginary artifacts
432 syy[0] = 0
433 real = np.random.randn(*syy.shape)
434 imag = np.random.randn(*syy.shape)
435 epsilon = real + 1j * imag
436 rand = epsilon * np.sqrt(syy)
437 Y = np.real(np.fft.ifftn(rand * ntot))
438 Y = Y + mean
439 return Y
